Insertion Sort is great for small sized arrays. Θ(n2).
The algorithm is same that you would use to sort a hand of cards

function insertionSort(arr){

    // start at 2nd index
    // and build sorted array
    for(var j=1; j < arr.length; j++){

    var current = arr[j];
         var i = j-1;
         // insert a[j] into sorted array
         while(i >= 0 && arr[i] > current){
             arr[i+1] = arr[i];
             i--;
         }
     arr[i+1] = current;
    }
}
